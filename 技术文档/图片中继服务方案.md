# 图片中继服务方案

## 1. 架构设计

### 1.1 整体流程
```
客户端 → 云服务器 → 橘子API
  ↓         ↓          ↓
图片    存储+URL    视频生成
```

### 1.2 优势
- ✅ 绕过防火墙限制（小的URL vs 大的base64）
- ✅ 提高传输效率（URL只有几十字节）
- ✅ 图片可复用（同一图片不需要重复上传）
- ✅ 支持大图片（云服务器存储无限制）
- ✅ 可以添加图片处理（压缩、格式转换等）

```javascript
const express = require('express');
const multer = require('multer');
const { v4: uuidv4 } = require('uuid');
const sharp = require('sharp');
const path = require('path');
const fs = require('fs').promises;
const cors = require('cors');

const app = express();
const PORT = 3000;

// 配置
const UPLOAD_DIR = '/var/www/images';
const BASE_URL = 'https://your-server.com';
const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10MB
const EXPIRE_HOURS = 24;

// 中间件
app.use(cors());
app.use(express.json({ limit: '50mb' }));
app.use('/images', express.static(UPLOAD_DIR));

// 存储配置
const storage = multer.diskStorage({
  destination: async (req, file, cb) => {
    const date = new Date();
    const dir = path.join(
      UPLOAD_DIR,
      date.getFullYear().toString(),
      (date.getMonth() + 1).toString().padStart(2, '0'),
      date.getDate().toString().padStart(2, '0')
    );
    await fs.mkdir(dir, { recursive: true });
    cb(null, dir);
  },
  filename: (req, file, cb) => {
    const ext = path.extname(file.originalname);
    cb(null, `${uuidv4()}${ext}`);
  }
});

const upload = multer({
  storage,
  limits: { fileSize: MAX_FILE_SIZE },
  fileFilter: (req, file, cb) => {
    const allowedTypes = ['image/jpeg', 'image/png', 'image/webp'];
    if (allowedTypes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('只支持 JPEG, PNG, WebP 格式'));
    }
  }
});

// 图片压缩处理
async function compressImage(filePath, maxWidth = 1920) {
  try {
    const image = sharp(filePath);
    const metadata = await image.metadata();

    if (metadata.width > maxWidth) {
      await image
        .resize(maxWidth, null, { withoutEnlargement: true })
        .jpeg({ quality: 85 })
        .toFile(filePath + '.tmp');

      await fs.rename(filePath + '.tmp', filePath);
    }
  } catch (error) {
    console.error('图片压缩失败:', error);
  }
}

// 单文件上传
app.post('/api/upload', upload.single('file'), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ code: 400, msg: '未上传文件' });
    }

    // 可选：压缩图片
    if (req.query.compress === 'true') {
      await compressImage(req.file.path);
    }

    const date = new Date();
    const relativePath = path.join(
      date.getFullYear().toString(),
      (date.getMonth() + 1).toString().padStart(2, '0'),
      date.getDate().toString().padStart(2, '0'),
      req.file.filename
    );

    const url = `${BASE_URL}/images/${relativePath.replace(/\\/g, '/')}`;

    res.json({
      code: 200,
      msg: 'success',
      data: {
        url,
        filename: req.file.filename,
        size: req.file.size,
        expire_at: Math.floor(Date.now() / 1000) + EXPIRE_HOURS * 3600
      }
    });
  } catch (error) {
    console.error('上传失败:', error);
    res.status(500).json({ code: 500, msg: error.message });
  }
});

// 批量上传
app.post('/api/upload/batch', upload.array('files', 10), async (req, res) => {
  try {
    if (!req.files || req.files.length === 0) {
      return res.status(400).json({ code: 400, msg: '未上传文件' });
    }

    const urls = req.files.map(file => {
      const date = new Date();
      const relativePath = path.join(
        date.getFullYear().toString(),
        (date.getMonth() + 1).toString().padStart(2, '0'),
        date.getDate().toString().padStart(2, '0'),
        file.filename
      );
      return `${BASE_URL}/images/${relativePath.replace(/\\/g, '/')}`;
    });

    res.json({
      code: 200,
      msg: 'success',
      data: { urls }
    });
  } catch (error) {
    console.error('批量上传失败:', error);
    res.status(500).json({ code: 500, msg: error.message });
  }
});

// Base64上传
app.post('/api/upload/base64', async (req, res) => {
  try {
    const { image } = req.body;
    if (!image) {
      return res.status(400).json({ code: 400, msg: '缺少image参数' });
    }

    // 解析base64
    const matches = image.match(/^data:image\/(\w+);base64,(.+)$/);
    if (!matches) {
      return res.status(400).json({ code: 400, msg: '无效的base64格式' });
    }

    const ext = matches[1];
    const base64Data = matches[2];
    const buffer = Buffer.from(base64Data, 'base64');

    // 检查大小
    if (buffer.length > MAX_FILE_SIZE) {
      return res.status(400).json({ code: 400, msg: '文件过大' });
    }

    // 生成文件路径
    const date = new Date();
    const dir = path.join(
      UPLOAD_DIR,
      date.getFullYear().toString(),
      (date.getMonth() + 1).toString().padStart(2, '0'),
      date.getDate().toString().padStart(2, '0')
    );
    await fs.mkdir(dir, { recursive: true });

    const filename = `${uuidv4()}.${ext}`;
    const filePath = path.join(dir, filename);

    // 保存文件
    await fs.writeFile(filePath, buffer);

    const relativePath = path.join(
      date.getFullYear().toString(),
      (date.getMonth() + 1).toString().padStart(2, '0'),
      date.getDate().toString().padStart(2, '0'),
      filename
    );

    const url = `${BASE_URL}/images/${relativePath.replace(/\\/g, '/')}`;

    res.json({
      code: 200,
      msg: 'success',
      data: { url }
    });
  } catch (error) {
    console.error('Base64上传失败:', error);
    res.status(500).json({ code: 500, msg: error.message });
  }
});

// 健康检查
app.get('/health', (req, res) => {
  res.json({ status: 'ok', timestamp: Date.now() });
});

// 定期清理过期文件（每小时执行一次）
setInterval(async () => {
  try {
    const now = Date.now();
    const expireTime = EXPIRE_HOURS * 3600 * 1000;

    async function cleanDir(dir) {
      const files = await fs.readdir(dir, { withFileTypes: true });
      for (const file of files) {
        const fullPath = path.join(dir, file.name);
        if (file.isDirectory()) {
          await cleanDir(fullPath);
        } else {
          const stats = await fs.stat(fullPath);
          if (now - stats.mtimeMs > expireTime) {
            await fs.unlink(fullPath);
            console.log('删除过期文件:', fullPath);
          }
        }
      }
    }

    await cleanDir(UPLOAD_DIR);
  } catch (error) {
    console.error('清理失败:', error);
  }
}, 3600000);

app.listen(PORT, () => {
  console.log(`图片中继服务运行在 http://localhost:${PORT}`);
});
```

from flask import Flask, request, jsonify, send_from_directory
from werkzeug.utils import secure_filename
import os
import uuid
from datetime import datetime, timedelta
from PIL import Image
import base64
import io

app = Flask(__name__)

# 配置
UPLOAD_DIR = '/var/www/images'
BASE_URL = 'https://your-server.com'
MAX_FILE_SIZE = 10 * 1024 * 1024
ALLOWED_EXTENSIONS = {'png', 'jpg', 'jpeg', 'webp'}

def allowed_file(filename):
    return '.' in filename and filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

def get_upload_path():
    now = datetime.now()
    dir_path = os.path.join(
        UPLOAD_DIR,
        str(now.year),
        str(now.month).zfill(2),
        str(now.day).zfill(2)
    )
    os.makedirs(dir_path, exist_ok=True)
    return dir_path

@app.route('/api/upload', methods=['POST'])
def upload_file():
    if 'file' not in request.files:
        return jsonify({'code': 400, 'msg': '未上传文件'}), 400

    file = request.files['file']
    if file.filename == '':
        return jsonify({'code': 400, 'msg': '文件名为空'}), 400

    if not allowed_file(file.filename):
        return jsonify({'code': 400, 'msg': '不支持的文件格式'}), 400

    # 生成唯一文件名
    ext = file.filename.rsplit('.', 1)[1].lower()
    filename = f"{uuid.uuid4()}.{ext}"

    # 保存文件
    upload_path = get_upload_path()
    filepath = os.path.join(upload_path, filename)
    file.save(filepath)

    # 可选：压缩图片
    if request.args.get('compress') == 'true':
        img = Image.open(filepath)
        if img.width > 1920:
            ratio = 1920 / img.width
            new_size = (1920, int(img.height * ratio))
            img = img.resize(new_size, Image.LANCZOS)
            img.save(filepath, quality=85, optimize=True)

    # 生成URL
    now = datetime.now()
    relative_path = f"{now.year}/{str(now.month).zfill(2)}/{str(now.day).zfill(2)}/{filename}"
    url = f"{BASE_URL}/images/{relative_path}"

    return jsonify({
        'code': 200,
        'msg': 'success',
        'data': {
            'url': url,
            'filename': filename,
            'size': os.path.getsize(filepath)
        }
    })

@app.route('/api/upload/base64', methods=['POST'])
def upload_base64():
    data = request.get_json()
    if not data or 'image' not in data:
        return jsonify({'code': 400, 'msg': '缺少image参数'}), 400

    image_data = data['image']

    # 解析base64
    if image_data.startswith('data:image'):
        header, base64_data = image_data.split(',', 1)
        ext = header.split('/')[1].split(';')[0]
    else:
        base64_data = image_data
        ext = 'png'

    # 解码
    image_bytes = base64.b64decode(base64_data)

    # 生成文件名
    filename = f"{uuid.uuid4()}.{ext}"
    upload_path = get_upload_path()
    filepath = os.path.join(upload_path, filename)

    # 保存
    with open(filepath, 'wb') as f:
        f.write(image_bytes)

    # 生成URL
    now = datetime.now()
    relative_path = f"{now.year}/{str(now.month).zfill(2)}/{str(now.day).zfill(2)}/{filename}"
    url = f"{BASE_URL}/images/{relative_path}"

    return jsonify({
        'code': 200,
        'msg': 'success',
        'data': {'url': url}
    })

@app.route('/images/<path:filename>')
def serve_image(filename):
    return send_from_directory(UPLOAD_DIR, filename)

@app.route('/health')
def health():
    return jsonify({'status': 'ok', 'timestamp': datetime.now().timestamp()})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=3000)
```

// lib/services/image_relay_service.dart
import 'dart:convert';
import 'dart:io';
import 'package:dio/dio.dart';
import 'package:path/path.dart' as path;

class ImageRelayService {
  final String relayServerUrl;
  final Dio _dio;

  ImageRelayService({required this.relayServerUrl}) : _dio = Dio();

  /// 上传base64图片到中继服务器
  Future<String> uploadBase64Image(String base64Image) async {
    try {
      final response = await _dio.post(
        '$relayServerUrl/api/upload/base64',
        data: {'image': 'data:image/png;base64,$base64Image'},
      );

      if (response.data['code'] == 200) {
        return response.data['data']['url'];
      } else {
        throw Exception(response.data['msg'] ?? '上传失败');
      }
    } catch (e) {
      throw Exception('上传到中继服务器失败: $e');
    }
  }

  /// 批量上传base64图片
  Future<List<String>> uploadBase64Images(List<String> base64Images) async {
    final urls = <String>[];
    for (final img in base64Images) {
      final url = await uploadBase64Image(img);
      urls.add(url);
    }
    return urls;
  }

  /// 上传文件
  Future<String> uploadFile(File file) async {
    try {
      final formData = FormData.fromMap({
        'file': await MultipartFile.fromFile(
          file.path,
          filename: path.basename(file.path),
        ),
      });

      final response = await _dio.post(
        '$relayServerUrl/api/upload',
        data: formData,
      );

      if (response.data['code'] == 200) {
        return response.data['data']['url'];
      } else {
        throw Exception(response.data['msg'] ?? '上传失败');
      }
    } catch (e) {
      throw Exception('上传文件失败: $e');
    }
  }
}
```
